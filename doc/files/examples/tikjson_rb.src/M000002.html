def json_reply(response)
  return '[' +
  response.map do |sentence| 
    '{' +
    sentence.map do |key, value|
      json_str(key) + ': ' +
      if value.nil?
        'NULL'
      elsif /^-?(?:\d+(\.\d+)?|\d*\.\d+)$/.match(value)
        value
      else
        json_str(value)
      end
    end.join(',') +
    '}'
  end.join(',') +
  ']'
end
